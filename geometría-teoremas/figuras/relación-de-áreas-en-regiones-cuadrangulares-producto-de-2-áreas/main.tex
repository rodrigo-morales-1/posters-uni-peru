\coordinate (D) at (5, 0);

    (A) node[below left]{$A$}
    -- (B) node[above left]{$B$}

\draw [dashed] (B) -- (D);

\node at (barycentric cs:A=1,B=1,P=1) {$S_1$};
\node at (barycentric cs:B=1,C=1,P=1) {$S_2$};
